<div id="cur-dot"></div>

<script>
  function openM(){document.getElementById('mob').classList.add('on');document.body.style.overflow='hidden'}
  function closeM(){document.getElementById('mob').classList.remove('on');document.body.style.overflow=''}
  const obs=new IntersectionObserver(e=>{e.forEach(el=>{if(el.isIntersecting)el.target.classList.add('on')})},{threshold:0.08});
  document.querySelectorAll('.r').forEach(el=>obs.observe(el));
  window.addEventListener('scroll',()=>{document.querySelector('header').style.borderBottomColor=window.scrollY>20?'rgba(201,168,76,0.3)':'rgba(201,168,76,0.18)'});

  // SCROLL SCISSORS
  const ss = document.getElementById('scrollScissors');
  const trail = document.getElementById('cutTrail');
  const ss1 = document.getElementById('ss1');
  const ss2 = document.getElementById('ss2');
  let lastScroll = 0;
  let ssAngle = 0;
  let snipInterval;

  function updateScrollScissors(){
    const scrolled = window.scrollY;
    const total = document.body.scrollHeight - window.innerHeight;
    const pct = scrolled / total;

    // Show after scrolling a bit
    if(scrolled > 200){
      ss.classList.add('visible');
    } else {
      ss.classList.remove('visible');
    }

    // Position scissors vertically based on scroll %
    const topPx = pct * (window.innerHeight - 80) + 40;
    ss.style.top = topPx + 'px';
    ss.style.transform = 'translateY(0)';

    // Cut trail from top to scissors position
    trail.style.height = topPx + 'px';
    trail.style.top = '0px';

    // Snip animation when scrolling
    const delta = scrolled - lastScroll;
    if(Math.abs(delta) > 2){
      ss1.style.transition = 'transform 0.08s ease';
      ss2.style.transition = 'transform 0.08s ease';
      ss1.style.transform = 'rotate(-15deg)';
      ss2.style.transform = 'rotate(15deg)';
      clearTimeout(snipInterval);
      snipInterval = setTimeout(() => {
        ss1.style.transform = '';
        ss2.style.transform = '';
      }, 120);
    }
    lastScroll = scrolled;
  }

  window.addEventListener('scroll', updateScrollScissors);
  updateScrollScissors();

  // LIVE CLOCK
  function updateClock(){
    const now = new Date();
    let h = now.getHours(), m = now.getMinutes(), s = now.getSeconds();
    const ampm = h >= 12 ? 'PM' : 'AM';
    h = h % 12 || 12;
    const el = document.getElementById('liveClock');
    if(el) el.textContent = `${h}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')} ${ampm}`;
  }
  updateClock();
  setInterval(updateClock, 1000);

  // ANIMATE HOURS BARS ON SCROLL
  const barsObs = new IntersectionObserver(entries => {
    entries.forEach(e => {
      if(e.isIntersecting){
        setTimeout(() => {
          document.querySelectorAll('.hanim-fill').forEach((bar, i) => {
            setTimeout(() => bar.classList.add('loaded'), i * 200);
          });
        }, 300);
        barsObs.disconnect();
      }
    });
  }, {threshold: 0.3});
  const hoursWrap = document.querySelector('.hours-anim-wrap');
  if(hoursWrap) barsObs.observe(hoursWrap);
  const cur = document.getElementById('cur');
  const dot = document.getElementById('cur-dot');
  const b1 = document.getElementById('b1');
  const b2 = document.getElementById('b2');

  let mx=0, my=0, cx=0, cy=0;
  let snipping = false;

  // Smooth follow
  document.addEventListener('mousemove', e => {
    mx = e.clientX;
    my = e.clientY;
    dot.style.transform = `translate(${mx-2.5}px,${my-2.5}px)`;
  });

  // Smooth lerp animation
  function lerp(a,b,n){ return (1-n)*a + n*b; }
  function animCursor(){
    cx = lerp(cx, mx, 0.18);
    cy = lerp(cy, my, 0.18);
    cur.style.transform = `translate(${cx-12}px,${cy-12}px)`;
    requestAnimationFrame(animCursor);
  }
  animCursor();

  // Snip on click
  document.addEventListener('mousedown', () => {
    snipping = true;
    cur.classList.add('snip');
    b1.style.transform = 'rotate(-22deg)';
    b2.style.transform = 'rotate(22deg)';
  });
  document.addEventListener('mouseup', () => {
    snipping = false;
    cur.classList.remove('snip');
    b1.style.transform = '';
    b2.style.transform = '';
  });

  // Idle snip loop every 4 seconds
  setInterval(() => {
    if(snipping) return;
    b1.style.transition = 'transform 0.15s ease';
    b2.style.transition = 'transform 0.15s ease';
    b1.style.transform = 'rotate(-18deg)';
    b2.style.transform = 'rotate(18deg)';
    setTimeout(() => {
      b1.style.transform = '';
      b2.style.transform = '';
    }, 220);
  }, 4000);

  // Hover effect on interactive elements
  const hoverEls = document.querySelectorAll('a, button, .svc-card, .rev-card, .why-stat');
  hoverEls.forEach(el => {
    el.addEventListener('mouseenter', () => cur.classList.add('hover'));
    el.addEventListener('mouseleave', () => cur.classList.remove('hover'));
  });

  /* ═════ NEW PREMIUM INJECTED JS ═════ */
  
  // 1. Generate Gold Dust Particles in Hero
  const heroSec = document.querySelector('.hero');
  if(heroSec) {
    const particleContainer = document.createElement('div');
    particleContainer.id = 'particles';
    heroSec.insertBefore(particleContainer, heroSec.firstChild);

    for(let i = 0; i < 35; i++) {
      let p = document.createElement('div');
      p.className = 'dust';
      let size = Math.random() * 3 + 1;
      p.style.width = size + 'px';
      p.style.height = size + 'px';
      p.style.left = Math.random() * 100 + '%';
      p.style.animationDuration = (Math.random() * 6 + 4) + 's';
      p.style.animationDelay = (Math.random() * 5) + 's';
      p.style.opacity = Math.random() * 0.5 + 0.1;
      particleContainer.appendChild(p);
    }
  }

  // 2. Magnetic Button Hover Effect
  const magneticBtns = document.querySelectorAll('.btn-gold, .btn-ghost');
  magneticBtns.forEach(btn => {
    btn.addEventListener('mousemove', function(e) {
      const rect = this.getBoundingClientRect();
      const h = rect.width / 2;
      const v = rect.height / 2;
      const x = e.clientX - rect.left - h;
      const y = e.clientY - rect.top - v;
      // Pulls the button slightly toward the cursor
      this.style.transform = `translate(${x * 0.25}px, ${y * 0.25}px) scale(1.03)`;
    });
    btn.addEventListener('mouseleave', function() {
      this.style.transform = 'translate(0px, 0px) scale(1)';
    });
  });

  // 3. Smooth Parallax Hero Elements on Scroll
  window.addEventListener('scroll', () => {
    const scrolled = window.scrollY;
    const heroH1 = document.querySelector('.hero-h1');
    const heroRight = document.querySelector('.hero-right');
    
    // Only animate while in the top section to save performance
    if(scrolled < window.innerHeight) {
      if(heroH1) heroH1.style.transform = `translateY(${scrolled * 0.15}px)`;
      if(heroRight) heroRight.style.transform = `translateY(${scrolled * 0.08}px)`;
    }
  });

  // 4. Add Shimmer Class to Emphasized Texts
  document.querySelectorAll('.hero-h1 em, .sec-h2 em, .cta-h em').forEach(el => {
    el.classList.add('shimmer');
  });

</script>
